22        self._validate()

26    def _validate(self) -> None:

32    def _get(self) -> str | None:
33        """Get the current offset value.
34
35        Returns:
36            str: The current offset value.
37        """
38        resp = self.redis_client.get(self.key)
39        if resp is None:
40            return None
41
42        try:
43            decoded_response = resp.decode("UTF-8")
44        except ValueError as e:
45            self._log.error(
46                f"Got invalid offset response from Redis for key {self.key}: {resp}"
47            )
48            raise e
49        return decoded_response

52        """Set a new offset

58            Exception: If error is returned from Redis.
59
60        Returns:
61            bool: Whether set was successful.
62        """
63        resp = self.redis_client.set(self.key, value)
64        if resp is None:
65            raise Exception(
66                f"Got invalid response `None` from `redis.Redis.set()` for key '{self.key}'"
67            )

 11class PostgresOffsetManager(OffsetManager):
 12    """Manager for offset values stored in a Postgres database."""
 13
 14    def __init__(
 15        self,
 16        connection_pool: PostgresConnectionPool,
 17        schema: str,
 18        table: str,
 19        key_field: str,
 20        offset_field: str,
 21        offset_type: OffsetType | None,
 22    ) -> None:
 23        """Initialize the `PostgresOffsetManager`.
 24
 25        Args:
 26            connection_pool (PostgresConnectionPool): The connection pool to use.
 27            schema (str): The database schema name.
 28            table (str): The database table name.
 29            key_field (str): The field name which contains the offset key.
 30            offset_field (str): The field name which  contains the offset value.
 31            offset_type (PossibleOffsetType): Enum that represents the type of offset value. Defaults to "int".
 32        """
 33        super().__init__(offset_type=offset_type)
 34
 35        self.connection_pool = connection_pool
 36        self.schema = schema
 37        self.table = table
 38        self.key_field = key_field
 39        self.offset_field = offset_field
 40
 41        # validate offset data structure
 42        self._validate()
 43
 44    def _connect(*args) -> Callable:
 45        """Decorator function to get and put a Postgres connection from the connection pool and return a cursor."""
 46
 47        def wrap(f):
 48            @wraps(f)
 49            def wrapper(self, *args, **kwargs):
 50                try:
 51                    # Setup postgres connection
 52                    connection = self.connection_pool.getconn()
 53                    cursor = connection.cursor()
 54
 55                    # Call function passing in cursor
 56                    return_val = f(self, cursor, *args, **kwargs)
 57
 58                finally:
 59                    # Close connection
 60                    self.connection_pool.putconn(connection)
 61
 62                return return_val
 63
 64            return wrapper
 65
 66        return wrap
 67
 68    @_connect()
 69    def _validate(self, cursor) -> None:
 70        try:
 71            cursor.execute(
 72                SQL("SELECT 1 FROM {schema}.{table} LIMIT 1").format(
 73                    schema=Identifier(self.schema), table=Identifier(self.table)
 74                ),
 75            )
 76        except Exception as e:
 77            raise Exception(
 78                f"Error validating offset table {self.schema}.{self.table}: {e}"
 79            ) from e
 80
 81        try:
 82            cursor.execute(
 83                SQL("SELECT {key_field} FROM {schema}.{table} LIMIT 1").format(
 84                    key_field=Identifier(self.key_field),
 85                    schema=Identifier(self.schema),
 86                    table=Identifier(self.table),
 87                )
 88            )
 89        except Exception as e:
 90            raise Exception(
 91                f"Error validating offset key_field {self.key_field}: {e}"
 92            ) from e
 93
 94        try:
 95            cursor.execute(
 96                SQL("SELECT {offset_field} FROM {schema}.{table} LIMIT 1").format(
 97                    offset_field=Identifier(self.offset_field),
 98                    schema=Identifier(self.schema),
 99                    table=Identifier(self.table),
100                )
101            )
102        except Exception as e:
103            raise Exception(
104                f"Error validating offset_field {self.offset_field}: {e}"
105            ) from e
106        pass
107
108    @_connect()
109    def _get(self, cursor) -> str | None:
110        """Get the current offset value.
111
112        Returns:
113            str: The current offset value.
114        """
115        cursor.execute(
116            SQL(
117                "SELECT {offset_field} FROM {schema}.{table} WHERE {key_field} = %s"
118            ).format(
119                offset_field=Identifier(self.offset_field),
120                schema=Identifier(self.schema),
121                table=Identifier(self.table),
122                key_field=Identifier(self.key_field),
123            ),
124            [self.key],
125        )
126        offset = cursor.fetchone()
127        if offset is None:
128            return None
129
130        return str(offset[0])
131
132    @_connect()
133    def _set(self, cursor, value: str) -> bool:
134        """Set a new offset
135
136        Args:
137            value (str): Value to set to.
138
139        Raises:
140            Exception: If error is returned from Postgres.
141
142        Returns:
143            bool: Whether set was successful.
144        """
145        cursor.execute(
146            SQL(
147                "UPDATE {schema}.{table} SET {offset_field} = %s WHERE {key_field} = %s;"
148            ).format(
149                schema=Identifier(self.schema),
150                table=Identifier(self.table),
151                offset_field=Identifier(self.offset_field),
152                key_field=Identifier(self.key_field),
153            ),
154            [value, self.key],
155        )
156        if cursor.rowcount == 0:
157            cursor.execute(
158                SQL(
159                    "INSERT INTO {schema}.{table} ({key_field}, {offset_field}) VALUES(%s, %s);"
160                ).format(
161                    schema=Identifier(self.schema),
162                    table=Identifier(self.table),
163                    key_field=Identifier(self.key_field),
164                    offset_field=Identifier(self.offset_field),
165                ),
166                [self.key, value],
167            )
168        cursor.connection.commit()
169
170        return True
