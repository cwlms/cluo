 47        """Initialize the `KafkaSink`.
 48
 49        Args:
 50            bootstrap_servers (str, required): Initial list of Kafka brokers.
 51            topic (str, required): Name of the topic that messages will be passed to.
 52            flush_timeout (float, optional): Maximum time flushing messages will block.
 53            get_message_key (GetMessageKeyCallable, optional): A callback to generate a Kafka message key to ensure ordering. Defaults to returning None.
 54            name (str, optional): Stage name. Defaults to class name if name = None.
 55            processes (int, optional): Number of CPUs to use. Can only be used if .process_record method is implemented. Defaults to 1.
 56            expose_metrics (bool, optional): Whether or not to expose metrics for this sink. Defaults to False.
 57        """
 58        Sink.__init__(
 59            self, name=name, processes=processes, expose_metrics=expose_metrics
 60        )
 61
 62        self.conf = {
 63            "bootstrap.servers": bootstrap_servers,
 64            "transactional.id": uuid4(),  # TODO: self.unique_name()
 65        }
 66        self.producer: Optional[Producer] = None
 67
 68        self.topic = topic
 69        self.flush_timeout = flush_timeout
 70        self.get_message_key = get_message_key
 71        # used to ensure transactions are enforced
 72        self._inside_context = False
 73
 74    def _get_producer(self) -> Producer:
 75        if self.producer is None:
 76            self.producer = Producer(self.conf)
 77            try:
 78                self.producer.init_transactions()
 79            except KafkaError as exc:
 80                if exc.args[0].retriable():
 81                    self.producer.init_transactions()
 82                else:
 83                    raise exc
 84        return self.producer
 85
 86    def __enter__(self) -> None:
 87        self._inside_context = True
 88        if config_state.RUN_MODE == RunMode.WRITE:
 89            producer = self._get_producer()
 90            try:
 91                producer.begin_transaction()
 92            except KafkaError as exc:
 93                if exc.args[0].retriable():
 94                    producer.begin_transaction()
 95                else:
 96                    raise exc
 97
 98    def __exit__(
 99        self,
100        exc_type: Type[BaseException] | None,
101        exc_value: BaseException | None,
102        exc_tb: TracebackType | None,
103    ) -> bool:
104        self._inside_context = False
