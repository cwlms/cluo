118    def validate_pipeline(self) -> None:
119        """Validates a pipeline and raises an error if it does not pass.
120
121        Currently we check for the following things when validating:
122        * cycles in the graph.
123        * channels which have no downstreams
124        * stages which are not sinks and have no downstreams

126        """

146    def _validate_no_cycles(self) -> None:
147        """Raise an error if there is a cycle in the pipeline graph."""

152    def _validate_no_unmonitored_channels(self) -> None:
153        """Raise and error if there is a channel with no downstreams on a non-sink."""
154        stage_queue: list[Stage] = list(self.sources)
155        while stage_queue:
156            cur_stage = stage_queue.pop(0)
157            channels = cur_stage.channels
158            for channel in channels:
159                channel_downstreams = cur_stage.get_downstreams(channel=channel)
160                unmonitored_flag = len(channel_downstreams) == 0 and not isinstance(
161                    cur_stage, Sink
162                )
163
164                if unmonitored_flag:
165                    raise PipelineValidationError(
166                        f"Stage {cur_stage} is not a sink and {channel} does not have any downstream stages."
167                    )
168
169                stage_queue.extend(iter(channel_downstreams))

171    def generate_pipeline_graph(self) -> DiGraph:
172        graph = DiGraph()
173        stage_queue: list[Stage] = list(self.sources)
174        added_stages = set()
175        while stage_queue:
176            current_stage = stage_queue.pop(0)
177            if current_stage.unique_name not in graph.nodes:
178                graph.add_node(current_stage.unique_name)
179                added_stages.add(current_stage)
180            for channel in current_stage.channels:
181                channel_stages = current_stage.get_downstreams(channel=channel)
182                for channel_stage in channel_stages:
183                    if channel_stage not in added_stages:
184                        stage_queue.append(channel_stage)
185                        graph.add_node(channel_stage.unique_name)
186                        added_stages.add(channel_stage)
187                    graph.add_edge(current_stage.unique_name, channel_stage.unique_name)
188        return graph

190    @_validate
191    def run(self) -> None:
192        """Run the pipeline. Enters `while True` loop and runs all stages in pipeline."""
193        while True:
194            __start = time.time()
195            for source in self.sources:
196                with source:
197                    self._single_run(source, Batch([{}]))  # type: ignore
198            __end = time.time()
199            __wait = self.interval - (__end - __start)
200            if __wait > 0:
201                time.sleep(__wait)
202            if self.expose_metrics:
203                self.runs_processed.labels(pipeline_name=self.name).inc()
204                self.pipeline_processing_time.labels(pipeline_name=self.name).set(
205                    __end - __start
206                )

208    @_validate
209    def _single_run(self, stage: Stage, batch: Batch) -> None:
210        if (
211            len(batch) == 0
212            and not isinstance(stage, Source)
213            and not stage.process_empty_batches
214        ):
215            return
216        batch = stage(batch)
217
218        valid_batch, errored_batch = self._filter_errored_records(batch=batch)
219
220        self._process_errored_records(batch=errored_batch)
221
222        self._process_valid_records(stage=stage, batch=valid_batch)

224    @staticmethod
225    def _filter_errored_records(batch: Batch) -> tuple[Batch, Batch]:
226        """Filter out errored records from valid records.
227
228        Args:
229            batch (Batch): Batch to be filtered.
230
231        Returns:
232            valid_batch (Batch): Batch with the good records.
233            errored_batch (Batch): Batch with the errored records.
234        """
235        errored_batch = Batch()
236        valid_batch = Batch()
237        for record in batch.records:
238            if record.error_count == 0:
239                valid_batch.records.append(record)
240            else:
241                errored_batch.records.append(record)
242        return valid_batch, errored_batch

244    def _process_errored_records(self, batch: Batch) -> None:
